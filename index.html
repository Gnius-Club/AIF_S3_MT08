def connect_wifi():

    wlan.active(True)

    if not wlan.isconnected():

        display.scroll("WIFI...")

            display.show(Image.ASLEEP)

            time.sleep_ms(300)

            display.clear()

            time.sleep_ms(300)

            timeout += 1

    if wlan.isconnected():

        print('Conexi√≥n Wi-Fi exitosa:', wlan.ifconfig())

        return False
